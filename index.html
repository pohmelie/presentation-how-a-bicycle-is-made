Iterator

 1 class Countdown:
 2     def __init__(self, n):
 3         self.n = n
 4 
 5     def __iter__(self):
 6         return self
 7 
 8     def __next__(self):
 9         if self.n == 0:
10             raise StopIteration()
11         v = self.n
12         self.n -= 1
13         return v
